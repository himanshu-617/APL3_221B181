// Pandav Class (inherits Bharatvanshi)
class Pandav extends Bharatvanshi {
    @Override
    void fight() {
        System.out.println("Pandav fights bravely.");
    }

    @Override
    void obey() {
        System.out.println("Pandav is obedient.");
    }

    @Override
    void kind() {
        System.out.println("Pandav is kind.");
    }
}
