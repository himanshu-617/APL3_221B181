// Kaurav Class (inherits Bharatvanshi)
class Kaurav extends Bharatvanshi {
    @Override
    void fight() {
        System.out.println("Kaurav fights aggressively.");
    }

    @Override
    void obey() {
        System.out.println("Kaurav is disobedient.");
    }

    @Override
    void kind() {
        System.out.println("Kaurav is cruel.");
    }
}
